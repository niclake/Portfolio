SELECT CompanyName, ProductName, CategoryName, SUM(Quantity) AS NumOrdered
FROM Suppliers, Products, Categories, Order_Details
WHERE Suppliers.SupplierID = Products.SupplierID
AND Categories.CategoryID = Products.CategoryID
AND Products.ProductID = Order_Details.ProductID
GROUP BY Products.ProductID
ORDER BY NumOrdered DESC
